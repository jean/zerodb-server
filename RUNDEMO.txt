#
# Start the stunnel server from ZEOServer and the stunnel client from
# zerodb.DB. Applications must provide stunnel configuration files.
#

git clone git@github.com:zero-db/zerodb
cd zerodb
git checkout stunnel-builtin
cd ..

git clone git@github.com:zero-db/zerodb-server
cd zerodb-server
git checkout stunnel-builtin-demo
virtualenv-2.7 .

./bin/pip install -e .
./bin/pip install -e ../zerodb
./bin/pip install -r demo/requirements.txt

./bin/zerodb-manage init_db \
	--username=root \
	--passphrase=123 \
	--sock=var/zerodb-server.sock \
	--stunnel-server=localhost:9001 \
	--stunnel-client=localhost:8001

First terminal:
./bin/zerodb-server

Second terminal:
./bin/python demo/create.py
./bin/python demo/demo.py
